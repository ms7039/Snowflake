CREATE DATABASE IF NOT EXISTS RETAIL_DB;

CREATE SCHEMA IF NOT EXISTS RETAIL_DB.ORDERS_DEMO;

CREATE WAREHOUSE IF NOT EXISTS RETAIL_LOAD_WH
    WAREHOUSE_SIZE = 'X-SMALL'
    AUTO_SUSPEND = 60
    AUTO_RESUME = TRUE
    COMMENT = 'Dedicated warehouse for retail data loading';

CREATE ROLE IF NOT EXISTS RETAIL_DATA_ENGINEER_ROLE;

CREATE USER IF NOT EXISTS RETAIL_ENGINEER_USER
    PASSWORD = 'RetailEng@2026!'
    DEFAULT_ROLE = RETAIL_DATA_ENGINEER_ROLE
    DEFAULT_WAREHOUSE = RETAIL_LOAD_WH
    DEFAULT_NAMESPACE = 'RETAIL_DB.ORDERS_DEMO'
    MUST_CHANGE_PASSWORD = FALSE;

GRANT USAGE ON WAREHOUSE RETAIL_LOAD_WH TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT OPERATE ON WAREHOUSE RETAIL_LOAD_WH TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT USAGE ON DATABASE RETAIL_DB TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT USAGE ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE TABLE ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE STAGE ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE PIPE ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE STREAM ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE TASK ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT CREATE FILE FORMAT ON SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT ALL ON ALL TABLES IN SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;
GRANT ALL ON FUTURE TABLES IN SCHEMA RETAIL_DB.ORDERS_DEMO TO ROLE RETAIL_DATA_ENGINEER_ROLE;

GRANT ROLE RETAIL_DATA_ENGINEER_ROLE TO USER RETAIL_ENGINEER_USER;
GRANT ROLE RETAIL_DATA_ENGINEER_ROLE TO USER MS7039;

GRANT EXECUTE TASK ON ACCOUNT TO ROLE RETAIL_DATA_ENGINEER_ROLE;

SHOW DATABASES;
SHOW SCHEMAS;
SHOW WAREHOUSES;
SHOW ROLES;
SHOW USERS;
