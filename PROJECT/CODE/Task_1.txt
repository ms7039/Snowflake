CREATE WAREHOUSE IF NOT EXISTS HEALTHCARE_WH
    WAREHOUSE_SIZE = 'XSMALL'
    AUTO_SUSPEND = 60
    AUTO_RESUME = TRUE
    INITIALLY_SUSPENDED = FALSE;

CREATE DATABASE IF NOT EXISTS HEALTHCARE_DB;

CREATE SCHEMA IF NOT EXISTS APPOINTMENT_SCHEMA;

CREATE FILE FORMAT IF NOT EXISTS CSV_FORMAT
    TYPE = 'CSV'
    FIELD_OPTIONALLY_ENCLOSED_BY = '"'
    SKIP_HEADER = 1
    NULL_IF = ('NULL', 'null', '')
    EMPTY_FIELD_AS_NULL = TRUE
    TRIM_SPACE = TRUE;

CREATE STAGE IF NOT EXISTS APPOINTMENT_STAGE
    FILE_FORMAT = CSV_FORMAT;

SHOW WAREHOUSES LIKE 'HEALTHCARE_WH';
SHOW DATABASES LIKE 'HEALTHCARE_DB';
SHOW SCHEMAS LIKE 'APPOINTMENT_SCHEMA';
SHOW FILE FORMATS LIKE 'CSV_FORMAT';
SHOW STAGES LIKE 'APPOINTMENT_STAGE';
