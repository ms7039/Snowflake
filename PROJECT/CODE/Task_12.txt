USE WAREHOUSE HEALTHCARE_WH;
USE DATABASE HEALTHCARE_DB;
USE SCHEMA APPOINTMENT_SCHEMA;

CREATE OR REPLACE VIEW VW_APPOINTMENT_DASHBOARD AS
SELECT
    PATIENT_ID,
    APPOINTMENT_ID,
    GENDER,
    AGE,
    AGE_GROUP,
    SCHEDULED_DATE,
    APPOINTMENT_DATE,
    WAITING_DAYS,
    NEIGHBOURHOOD,
    SCHOLARSHIP,
    HYPERTENSION,
    DIABETES,
    ALCOHOLISM,
    HANDICAP,
    SMS_RECEIVED,
    NO_SHOW_STATUS,
    ATTENDANCE_STATUS,
    CASE WHEN NO_SHOW_STATUS = 'Yes' THEN 1 ELSE 0 END  AS IS_NO_SHOW,
    CASE WHEN NO_SHOW_STATUS = 'No'  THEN 1 ELSE 0 END  AS IS_ATTENDED,
    CASE WHEN SMS_RECEIVED   = 1     THEN 'SMS Sent' ELSE 'No SMS' END AS SMS_LABEL,
    YEAR(APPOINTMENT_DATE)  AS APPOINTMENT_YEAR,
    MONTH(APPOINTMENT_DATE) AS APPOINTMENT_MONTH,
    MONTHNAME(APPOINTMENT_DATE) AS APPOINTMENT_MONTH_NAME,
    DAYNAME(APPOINTMENT_DATE)   AS APPOINTMENT_DAY_NAME
FROM APPOINTMENT_FINAL
WHERE AGE >= 0
  AND APPOINTMENT_DATE IS NOT NULL
  AND SCHEDULED_DATE   IS NOT NULL;

SELECT COUNT(*) AS TOTAL_RECORDS FROM VW_APPOINTMENT_DASHBOARD;

SELECT
    AGE_GROUP,
    COUNT(*)                                    AS TOTAL_APPOINTMENTS,
    SUM(IS_NO_SHOW)                             AS NO_SHOW_COUNT,
    ROUND(SUM(IS_NO_SHOW) * 100.0 / COUNT(*), 2) AS NO_SHOW_RATE_PCT
FROM VW_APPOINTMENT_DASHBOARD
GROUP BY AGE_GROUP
ORDER BY NO_SHOW_RATE_PCT DESC;

SELECT
    SMS_LABEL,
    COUNT(*)                                    AS TOTAL_APPOINTMENTS,
    SUM(IS_NO_SHOW)                             AS NO_SHOW_COUNT,
    ROUND(SUM(IS_NO_SHOW) * 100.0 / COUNT(*), 2) AS NO_SHOW_RATE_PCT
FROM VW_APPOINTMENT_DASHBOARD
GROUP BY SMS_LABEL;

SELECT
    NEIGHBOURHOOD,
    COUNT(*)                                    AS TOTAL_APPOINTMENTS,
    SUM(IS_NO_SHOW)                             AS NO_SHOW_COUNT,
    ROUND(SUM(IS_NO_SHOW) * 100.0 / COUNT(*), 2) AS NO_SHOW_RATE_PCT
FROM VW_APPOINTMENT_DASHBOARD
GROUP BY NEIGHBOURHOOD
ORDER BY NO_SHOW_COUNT DESC
LIMIT 10;
