USE WAREHOUSE HEALTHCARE_WH;
USE DATABASE HEALTHCARE_DB;
USE SCHEMA APPOINTMENT_SCHEMA;

CREATE OR REPLACE STREAM APPOINTMENT_RAW_STREAM
ON TABLE APPOINTMENT_RAW
APPEND_ONLY = TRUE
COMMENT = 'Stream to track new rows inserted into APPOINTMENT_RAW';

SHOW STREAMS LIKE 'APPOINTMENT_RAW_STREAM';

SELECT SYSTEM$STREAM_HAS_DATA('APPOINTMENT_RAW_STREAM') AS HAS_NEW_DATA;
