MERGE INTO SALES_FINAL_TABLE AS tgt
USING (
    SELECT *
    FROM SALES_STREAM
    WHERE METADATA$ACTION   = 'INSERT'
      AND METADATA$ISUPDATE = TRUE
) AS src
ON tgt.id = src.id
WHEN MATCHED THEN
    UPDATE SET
        tgt.product  = src.product,
        tgt.price    = src.price,
        tgt.amount   = src.amount,
        tgt.store_id = src.store_id;
